const products = [ { id: 1, name: "p1", price: 34 }, { id: 2, name: "p2", price: 50 }, { id: 3, name: "p3", price: 75 } ]; const cart = { 1: 4, 3: 5 }; function add(products) { const cartItems = products.filter(p => cart[p.id]); let orderTot = cartItems.reduce((sum, item) => { const qty = cart[item.id]; const total = item.price * qty; console.log(`Name: ${item.name}, Price: ${item.price}, Qty: ${qty}, Total: ${total}`); return sum + total; }, 0); console.log("Order Total:", orderTot); } add(products);
